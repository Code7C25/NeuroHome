#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// Dirección I2C del LCD (comúnmente 0x27 o 0x3F)
LiquidCrystal_I2C lcd(0x27, 16, 2); // (Dirección, columnas, filas)

void setup() {
  lcd.init();              // Inicializar el LCD
  lcd.backlight();         // Encender la luz de fondo
  lcd.setCursor(0,0);      // Colocar cursor en columna 0, fila 0
  lcd.print("Hola Mundo!"); // Escribir en la primera fila
  lcd.setCursor(0,1);      // Colocar cursor en segunda fila
  lcd.print("Con Arduino :)");
}

  // Puedes poner aquí cualquier actualización de texto si lo deseas
}
