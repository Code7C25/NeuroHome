#include <SPI.h>
#include <MFRC522.h>

#define SS_PIN 10    // SDA del RC522 al pin 10 del Arduino
#define RST_PIN 9    // RST al pin 9 del Arduino

MFRC522 mfrc522(SS_PIN, RST_PIN);  // Crea el objeto del módulo

void setup() {
  Serial.begin(9600);
  SPI.begin();         // Inicia el bus SPI
  mfrc522.PCD_Init();  // Inicia el RC522

  Serial.println(F("Iniciando lector RFID RC522..."));
  Serial.println(F("Acerque su tarjeta al lector."));
  Serial.println();
}

void loop() {
  // Espera a que haya una nueva tarjeta
  if (!mfrc522.PICC_IsNewCardPresent()) return;

  // Selecciona una de las tarjetas
  if (!mfrc522.PICC_ReadCardSerial()) return;

  // Muestra el UID
  Serial.print(F("Tarjeta detectada! UID: "));
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    Serial.print(mfrc522.uid.uidByte[i] < 0x10 ? "0" : "");
    Serial.print(mfrc522.uid.uidByte[i], HEX);
    Serial.print(" ");
  }
  Serial.println();

  // Muestra tipo de tarjeta
  MFRC522::PICC_Type piccType = mfrc522.PICC_GetType(mfrc522.uid.sak);
  Serial.print(F("Tipo de tarjeta: "));
  Serial.println(mfrc522.PICC_GetTypeName(piccType));

  // Finaliza la lectura de la tarjeta actual
  mfrc522.PICC_HaltA();
  mfrc522.PCD_StopCrypto1();
}
